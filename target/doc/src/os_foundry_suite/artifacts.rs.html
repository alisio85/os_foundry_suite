1//! Workspace and artifact modeling.
2//!
3//! This module provides deterministic, side-effect free structures that describe:
4//!
5//! - where source code typically lives
6//! - where build artifacts are expected to be placed
7//! - how tools can refer to those artifacts
8//!
9//! The suite intentionally does not perform IO here. Downstream tooling can take these
10//! structures and implement filesystem operations.
11
12/// A deterministic workspace layout.
13///
14/// This describes a conventional directory structure that downstream projects can adopt.
15#[derive(Debug, Clone, PartialEq, Eq)]
16#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
17pub struct WorkspaceLayout {
18    /// Path to the kernel crate directory.
19    pub kernel_dir: String,
20
21    /// Path to the services directory.
22    pub services_dir: String,
23
24    /// Path to host tools directory.
25    pub tools_dir: String,
26
27    /// Path to build output directory.
28    pub out_dir: String,
29}
30
31impl WorkspaceLayout {
32    /// Returns a conventional workspace layout.
33    #[must_use]
34    pub fn conventional() -> Self {
35        Self {
36            kernel_dir: "kernel".to_string(),
37            services_dir: "services".to_string(),
38            tools_dir: "tools".to_string(),
39            out_dir: "out".to_string(),
40        }
41    }
42}
43
44/// Deterministic build metadata.
45///
46/// This structure is intentionally conservative. It avoids time-based fields by default
47/// to preserve reproducibility.
48#[derive(Debug, Clone, PartialEq, Eq)]
49#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
50pub struct BuildMetadata {
51    /// A stable build identifier.
52    ///
53    /// Downstream tooling may set this to a git hash, semver tag, or CI build number.
54    pub build_id: String,
55
56    /// Optional VCS revision string (e.g. git commit).
57    pub vcs_revision: Option<String>,
58}
59
60impl BuildMetadata {
61    /// Creates minimal build metadata with a human-supplied build id.
62    #[must_use]
63    pub fn new(build_id: impl Into<String>) -> Self {
64        Self {
65            build_id: build_id.into(),
66            vcs_revision: None,
67        }
68    }
69}
70
71/// A catalog of build artifacts.
72///
73/// Artifact paths are modeled as strings because:
74///
75/// - the suite aims to avoid filesystem dependencies
76/// - downstream tooling may run on different hosts
77#[derive(Debug, Clone, PartialEq, Eq)]
78#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
79pub struct ArtifactCatalog {
80    /// Build metadata.
81    pub meta: BuildMetadata,
82
83    /// Conventional workspace layout.
84    pub layout: WorkspaceLayout,
85
86    /// Optional path to the kernel binary.
87    pub kernel_binary: Option<String>,
88
89    /// Optional path to the produced image artifact.
90    pub image_artifact: Option<String>,
91
92    /// Optional path to a serial log or boot transcript.
93    pub boot_log: Option<String>,
94}
95
96impl ArtifactCatalog {
97    /// Creates an empty catalog.
98    #[must_use]
99    pub fn new(meta: BuildMetadata, layout: WorkspaceLayout) -> Self {
100        Self {
101            meta,
102            layout,
103            kernel_binary: None,
104            image_artifact: None,
105            boot_log: None,
106        }
107    }
108}

5/// A high-level boot artifact selection.
6///
7/// This enum is a suite-level intent signal describing what the boot chain should load.
8/// The suite itself does not implement loading.
9#[derive(Debug, Copy, Clone, PartialEq, Eq)]
10pub enum BootArtifact {
11    /// The kernel binary is loaded directly.
12    Kernel,
13
14    /// A UEFI application (EFI stub or bootloader) is loaded.
15    UefiApp,
16}
17
18/// Partition scheme selection.
19#[derive(Debug, Copy, Clone, PartialEq, Eq)]
20pub enum PartitionScheme {
21    /// No partitioning (single raw artifact).
22    None,
23
24    /// GPT partition scheme.
25    Gpt,
26}
27
28/// A suite-level partition descriptor.
29///
30/// This is intentionally minimal and stable.
31#[derive(Debug, Copy, Clone, PartialEq, Eq)]
32pub struct PartitionSpec {
33    /// A stable label for the partition.
34    pub label: &'static str,
35
36    /// Suggested size in MiB.
37    pub size_mib: u32,
38}
39
40/// Image layout intent.
41#[derive(Debug, Clone, PartialEq, Eq)]
42pub struct ImageLayoutIntent {
43    /// Partition scheme.
44    pub scheme: PartitionScheme,
45
46    /// Partitions.
47    pub partitions: Vec<PartitionSpec>,
48
49    /// Selected boot artifact intent.
50    pub boot: BootArtifact,
51}
52
53impl ImageLayoutIntent {
54    /// No layout.
55    #[must_use]
56    pub fn none() -> Self {
57        Self {
58            scheme: PartitionScheme::None,
59            partitions: Vec::new(),
60            boot: BootArtifact::Kernel,
61        }
62    }
63
64    /// A conservative UEFI + GPT layout intent.
65    #[must_use]
66    pub fn uefi_gpt_default() -> Self {
67        Self {
68            scheme: PartitionScheme::Gpt,
69            partitions: vec![
70                PartitionSpec {
71                    label: "ESP",
72                    size_mib: 128,
73                },
74                PartitionSpec {
75                    label: "OS",
76                    size_mib: 512,
77                },
78            ],
79            boot: BootArtifact::UefiApp,
80        }
81    }
82}

84/// A deterministic plan describing how an image should be packaged.
85///
86/// The plan is derived from blueprint intent and target selection.
87#[derive(Debug, Clone, PartialEq, Eq)]
88pub struct ImagePlan {
89    /// Kind of image.
90    pub kind: ImageKind,
91
92    /// Layout intent.
93    pub layout: ImageLayoutIntent,
94}
95
96/// Builds an [`ImagePlan`] from suite-level configuration.
97///
98/// This function is deterministic and side-effect free.
99pub fn plan_image(blueprint: &OsBlueprint, target: Target) -> Result<Option<ImagePlan>> {
100    validate_image_intent(&blueprint.image, blueprint.memory, target)?;
101
102    match blueprint.image.kind {
103        ImageKind::None => Ok(None),
104        ImageKind::Raw => Ok(Some(ImagePlan {
105            kind: ImageKind::Raw,
106            layout: ImageLayoutIntent::none(),
107        })),
108        ImageKind::UefiGpt => Ok(Some(ImagePlan {
109            kind: ImageKind::UefiGpt,
110            layout: ImageLayoutIntent::uefi_gpt_default(),
111        })),
112    }
113}
114
115fn validate_image_intent(image: &ImageConfig, memory: MemoryConfig, target: Target) -> Result<()> {
116    if image.kind == ImageKind::UefiGpt {
117        if !target.uefi {
118            return Err(Error::invalid_blueprint(
119                "image.kind is UefiGpt but target.uefi is false",
120            ));
121        }
122
123        if !memory.uefi_memory_map {
124            return Err(Error::invalid_blueprint(
125                "image.kind is UefiGpt but memory.uefi_memory_map is false",
126            ));
127        }
128    }
129
130    Ok(())

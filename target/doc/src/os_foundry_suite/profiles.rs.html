1use crate::config::{

6/// Suite-provided presets for quickly starting new OS projects.
7///
8/// Profiles are intentionally conservative and *host-runnable*: they only construct
9/// configuration values (they do not perform IO).
10///
11/// The goal is to offer strong defaults that are easy to override.
12pub struct Profiles;
13
14impl Profiles {
15    /// A minimal blueprint intended for small experiments.
16    #[must_use]
17    pub const fn minimal() -> OsBlueprint {
18        OsBlueprint {
19            name: "unnamed-os",
20            kernel: KernelConfig::minimal(),
21            memory: MemoryConfig::default_bare_metal(),
22            abi: AbiPolicy::strict(),
23            services: ServiceSet::minimal(),
24            observability: ObservabilityConfig::minimal(),
25            image: ImageConfig::none(),
26        }
27    }
28
29    /// A development blueprint intended for QEMU-based iteration.
30    #[must_use]
31    pub const fn dev_qemu() -> OsBlueprint {
32        OsBlueprint {
33            name: "unnamed-os",
34            kernel: KernelConfig::dev(),
35            memory: MemoryConfig::default_bare_metal(),
36            abi: AbiPolicy::strict(),
37            services: ServiceSet::dev(),
38            observability: ObservabilityConfig::dev(),
39            image: ImageConfig::raw(),
40        }
41    }
42
43    /// A development blueprint targeting UEFI workflows.
44    #[must_use]
45    pub const fn uefi_dev() -> OsBlueprint {
46        OsBlueprint {
47            name: "unnamed-os",
48            kernel: KernelConfig::dev(),
49            memory: MemoryConfig::default_uefi(),
50            abi: AbiPolicy::strict(),
51            services: ServiceSet::dev(),
52            observability: ObservabilityConfig::dev(),
53            image: ImageConfig::uefi_gpt(),
54        }
55    }
56}
